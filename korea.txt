    const toggleButton = document.getElementById('menu-toggle');
  const sidebar = document.querySelector('.sidebar');

  toggleButton.addEventListener('click', () => {
    sidebar.classList.toggle('open');
    document.body.classList.toggle('menu-open');

    const targetDate = new Date('2025-06-03T12:00:00-05:00');
    function updateCountdown() {
      const now = new Date();
      const distance = targetDate - now;
      const daysLeft = Math.ceil(distance / (1000 * 60 * 60 * 24));
     
      if (distance < 0) {
        document.getElementById("countdown").innerHTML = "Congratulations! A New Era Begins!";
      } else {
        document.getElementById("countdown").innerHTML = `D-${daysLeft}`;
      }
    }

    setInterval(updateCountdown, 1000 * 60 * 60);
    updateCountdown();
